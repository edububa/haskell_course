  \begin{minted}[bgcolor=bg]{haskell}
    factorial :: Num a => a -> a
    factorial 0 = 1
    factorial n = n * factorial (n - 1)
  \end{minted}

  \begin{minted}[bgcolor=bg]{haskell}
    duplicar :: Num a => [a] -> [a]
    duplicar []     = []
    duplicar [x]    = [x*2]
    duplicar (x:xs) = (x*2):(duplicar xs)
  \end{minted}
